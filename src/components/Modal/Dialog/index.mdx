## Dialog

### 基本用法

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false);
    return (
      <>
        <Button variant="primary" onClick={() => setIsOpen(true)}>
          Open
        </Button>
        <Dialog isOpen={isOpen} title="Title" onClose={() => setIsOpen(false)}>
          <p>Content</p>
        </Dialog>
      </>
    );
  }}
</Playground>

## Raw

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false);
    const onClose = () => setIsOpen(false);
    return (
      <>
        <Button variant="primary" onClick={() => setIsOpen(true)}>
          Open
        </Button>
        <Dialog.Container isOpen={isOpen} onClose={onClose}>
          <Dialog.Header onClickClose={onClose}>Title</Dialog.Header>
          <Dialog.Content>
            <p>Content</p>
          </Dialog.Content>
          <Dialog.Actions>
            <Button onClick={onClose}>取消</Button>
            <Button variant="primary" onClick={onClose}>
              确定
            </Button>
          </Dialog.Actions>
        </Dialog.Container>
      </>
    );
  }}
</Playground>
