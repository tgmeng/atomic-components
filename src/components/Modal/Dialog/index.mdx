# RawDialog

## 基本用法

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false);
    const onClose = () => setIsOpen(false);
    return (
      <>
        <Button variant="primary" onClick={() => setIsOpen(true)}>
          Open
        </Button>
        <RawDialog isOpen={isOpen} onClose={onClose}>
          <RawDialog.Header onClickClose={onClose}>Title</RawDialog.Header>
          <RawDialog.Content>
            <p>Content</p>
          </RawDialog.Content>
          <RawDialog.Actions>
            <Button onClick={onClose}>取消</Button>
            <Button variant="primary" onClick={onClose}>
              确定
            </Button>
          </RawDialog.Actions>
        </RawDialog>
      </>
    );
  }}
</Playground>

## 属性

<Props of={RawDialog} />
