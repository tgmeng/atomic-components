# Multiple modals

<Playground>

      <>
        <Button variant="primary" onClick={() => setIsOpen(true)}>

          </Button>
        </Modal>

          </Button>

          <Button onClick={() => setIsYetAnotherOpen(false)}>‍Close</Button>
        </Modal>
      </>
    );
  }}
</Playground>
