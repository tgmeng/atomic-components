# Modal

## 基本用法

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false);
    return (
      <>
        <Button variant="primary" onClick={() => setIsOpen(true)}>
          Open
        </Button>
        <Modal
          isOpen={isOpen}
          style={{ width: 500, height: 200 }}
          onClose={() => setIsOpen(false)}
        >
          <Button onClick={() => setIsOpen(false)}>‍Close</Button>
        </Modal>
      </>
    );
  }}
</Playground>

### Modal.Open 方法

<Playground>
  {() => {
    const render = ({ hasRandomSize = false } = {}) => (
      <div
        style={
          hasRandomSize
            ? {
                width: Math.floor(Math.random() * 400) + 100,
                height: Math.floor(Math.random() * 400) + 100,
              }
            : undefined
        }
      >
        <Button
          variant="primary"
          onClick={() =>
            Modal.open({
              children: render({ hasRandomSize: true }),
            })
          }
        >
          Open
        </Button>
      </div>
    );
    return render({ hasRandomSize: false });
  }}
</Playground>
