## 不同触发方式

<Playground>
  {() => {
    const [isOpen, setIsOpen] = React.useState(false);
    return (
      <>
        <Popper trigger="hover" content={<div>text</div>}>
          <Button>Hover me (default)</Button>
        </Popper>
        <Popper trigger="click" content={<div>text</div>}>
          <Button>Click me</Button>
        </Popper>
        <Popper trigger="focus" content={<div>text</div>}>
          <Button>Focus me</Button>
        </Popper>
        <Popper isOpen={isOpen} content={<div>text</div>}>
          <div>Manual</div>
        </Popper>
        <Button onClick={() => setIsOpen(true)}>Open</Button>
        <Button onClick={() => setIsOpen(false)}>Close</Button>
      </>
    );
  }}
</Playground>
